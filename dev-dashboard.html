Key Management Guidelines

For optimal security, follow these best practices:

  • **Rotate Keys Regularly:** Change your keys every 30-90 days, especially if multiple developers have access.
  • **Environment Variables:** Never hardcode API keys directly into your source code. Use environment variables (e.g., in `.env` files) for local development and secure secrets management services in production.
  • **Server-Side Usage:** Whenever possible, make API calls from your backend server to prevent exposing your keys on the client-side.
  • **Access Control:** Limit access to API keys only to individuals and systems that absolutely require them.
  • **IP Whitelisting:** If supported, restrict API key usage to specific IP addresses where your applications are hosted.

Python: Generate Basic Image

Quickly generate an image from a text prompt using Python.


import requests
import json

API_KEY = "YOUR_LUMIX_API_KEY"
API_ENDPOINT = "https://api.lumixai.com/v1/generate"

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "prompt": "A futuristic city skyline at sunset, cyberpunk style",
    "model": "Lumix-Gen-V3.1",
    "dimensions": "1024x1024",
    "num_images": 1
}

try:
    response = requests.post(API_ENDPOINT, headers=headers, json=data)
    response.raise_for_status() # Raise an exception for HTTP errors

    result = response.json()
    print("Generated Image URL:", result["data"][0]["url"])

except requests.exceptions.RequestException as e:
    print(f"API Request failed: {e}")
    print(f"Response: {response.text if 'response' in locals() else 'N/A'}")
Python, Image Generation Free

Node.js: Upscale Image

Upscale an existing image URL with Node.js.


const fetch = require('node-fetch');

const API_KEY = "YOUR_LUMIX_API_KEY";
const API_ENDPOINT = "https://api.lumixai.com/v1/upscale";

async function upscaleImage(imageUrl) {
    const headers = {
        "Authorization": `Bearer ${API_KEY}`,
        "Content-Type": "application/json"
    };

    const data = {
        "image_url": imageUrl,
        "scale": "2x",
        "model": "Lumix-Upscale-Pro"
    };

    try {
        const response = await fetch(API_ENDPOINT, {
            method: 'POST',
            headers: headers,
            body: JSON.stringify(data)
        });

        if (!response.ok) {
            const errorText = await response.text();
            throw new Error(`HTTP error! status: ${response.status}, ${errorText}`);
        }

        const result = await response.json();
        console.log("Upscaled Image URL:", result.data[0].url);

    } catch (error) {
        console.error("API Request failed:", error);
    }
}

upscaleImage("https://example.com/your-image.jpg");

PHP: Enhance Image

Improve image quality using the Lumix AI enhance endpoint in PHP.


<?php
$apiKey = "YOUR_LUMIX_API_KEY";
$apiEndpoint = "https://api.lumixai.com/v1/enhance";

$imageUrl = "https://example.com/low-res-image.jpg";

$headers = [
    "Authorization: Bearer " . $apiKey,
    "Content-Type: application/json"
];

$data = json_encode([
    "image_url" => $imageUrl,
    "quality_preset" => "high"
]);

$ch = curl_init($apiEndpoint);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, $data);

$response = curl_exec($ch);
$httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);

if (curl_errno($ch)) {
    echo 'Curl error: ' . curl_error($ch);
} elseif ($httpCode !== 200) {
    echo "API Error: " . $response;
} else {
    $result = json_decode($response, true);
    echo "Enhanced Image URL: " . $result['data'][0]['url'];
}

curl_close($ch);
?>

Node.js: Webhook Listener (Express)

Simple Express app to receive Lumix AI webhook events.


const express = require('express');
const bodyParser = require('body-parser');

const app = express();
const PORT = process.env.PORT || 3000;

// Use bodyParser to parse JSON bodies
app.use(bodyParser.json());

app.post('/lumix-webhook', (req, res) => {
    const event = req.body;
    console.log('Received Lumix AI Webhook Event:', event.event_type);
    console.log('Payload:', event.data);

    // Process the event based on its type
    if (event.event_type === 'generation.completed') {
        console.log('Image generation completed for:', event.data.image_id);
        // Add your logic here, e.g., update database, notify user
    } else if (event.event_type === 'billing.over_limit') {
        console.warn('Billing limit exceeded alert received!');
    }

    res.status(200).send('Webhook received successfully');
});

app.listen(PORT, () => {
    console.log(`Webhook listener running on port ${PORT}`);
    console.log(`Configure your Lumix AI webhook URL to: http://localhost:${PORT}/lumix-webhook`);
});

React.js: Display Generated Image

Conceptual React component to display an image generated by Lumix AI.


import React, { useState, useEffect } from 'react';

function LumixImageGenerator() {
    const [imageUrl, setImageUrl] = useState('');
    const [loading, setLoading] = useState(false);
    const API_ENDPOINT = "/api/generate-image"; // Your backend endpoint

    const generateImage = async () => {
        setLoading(true);
        try {
            // In a real app, API_KEY would be on your backend,
            // and this frontend calls your backend.
            const response = await fetch(API_ENDPOINT, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ prompt: "An abstract art piece" })
            });
            const data = await response.json();
            if (data.url) {
                setImageUrl(data.url);
            } else {
                console.error("No image URL received:", data);
            }
        } catch (error) {
            console.error("Error generating image:", error);
        } finally {
            setLoading(false);
        }
    };

    useEffect(() => {
        // Optional: generate an image on component mount
        // generateImage();
    }, []);

    return (
        <div>
            <button onClick={generateImage} disabled={loading}>
                {loading ? 'Generating...' : 'Generate New Image'}
            </button>
            {imageUrl && <img src={imageUrl} alt="Generated Lumix AI" style={{ maxWidth: '100%', marginTop: '20px' }} />}
            {!imageUrl && !loading && <p>Click 'Generate New Image' to create one.</p>}
        </div>
    );
}

export default LumixImageGenerator;
React.js, Frontend, UI Free
